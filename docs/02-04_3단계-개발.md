## 0. 이 단계에서 자주 막히는 포인트(먼저 체크)

✅ 이 문서에서 반드시 해결해주는 핵심 문제들입니다.

### 문제1) `products.html`의 헤더를 “동일”하게 유지하기 어려움

- “헤더/푸터를 복사”하는 방식이면 수정 시 모든 파일을 고쳐야 합니다.
- 본 실습은 **정적 HTML 기반**이므로 일단 **복사 방식으로 진행**하되, “실무 팁”으로 공통화 방향도 안내합니다.

### 문제2) `products.js`에서 생성된 카드에 이벤트가 안 먹는 경우

- `innerHTML`로 렌더링하면 기존 DOM이 갈아끼워져서 이벤트가 사라질 수 있습니다.
- 해결: **이벤트 위임(Event Delegation)** 적용

### 문제3) `main.js`에서 `handleSearch`가 정의되지 않음

- 현재 제공 코드에는 `handleSearch()`가 없음
- 문서에서 검색 동작을 **완성 코드로 추가**합니다.

### 문제4) `utils.js`의 `products`, `cart` 변수가 다른 파일에서 접근 불가

- `<script>` 로딩 순서/스코프 때문에 문제가 생깁니다.
- 해결: `utils.js`를 먼저 로드하거나, 전역에 안전하게 노출하는 방식으로 정리합니다.

## 3. CSS 스타일링

현재 `style.css`가 “페이지 전체 스타일 + 컴포넌트”가 섞여 있습니다.  
문서 구조상 `components.css`를 이미 만들었으니, 아래처럼 역할을 분리하면 강의자료로 더 좋습니다.

- `style.css`: Reset, root 변수, 레이아웃(container, header, footer), 페이지 섹션(hero 등)
- `components.css`: 버튼, 카드, 태그, 배지 등 재사용 컴포넌트
- `responsive.css`: 반응형 전용

✅ 보완 내용

- Pretendard를 쓸 거면 폰트 로드 필요(현재 HTML에 없음)

## 4. JavaScript 기능 구현 (가장 중요한 보완 파트)

### 4.1 `js/utils.js` (핵심 보완)

현재 `utils.js`는 “상품 데이터 + 카드 생성 + 장바구니”가 들어있는데, 실습에서 자주 발생하는 문제는 아래 3가지입니다.

- `products`, `cart`가 다른 파일에서 안 보임
- `cart` 변경 후 페이지 간 동기화가 어색함
- `add-to-cart` 클릭이 카드 클릭과 충돌(버블링)

✅ 해결 전략

- 전역 오염을 줄이면서도 “학습용으로 단순”하게: `window.App = {}` 네임스페이스 사용
- `cart`는 항상 `getCart()`/`setCart()`로 읽고 쓰기(동기화 안정)
- 카드 HTML에서 버튼에 `type="button"` / `data-action` 구분

아래는 교체 추천 `utils.js` 최종 예시입니다.

```javascript
/**
 * utils.js
 * - 공용 데이터(products)
 * - 공용 함수(카드 생성, cart 저장/로드)
 * - 다른 JS 파일에서 쓰기 쉽게 window.App 네임스페이스로 공개
 */

        <h3>${product.name}</h3>
        <div class="price">${App.formatPrice(product.price)}</div>

          class="btn btn-primary add-to-cart"
          data-action="add-to-cart"
          data-id="${product.id}"
        >
          장바구니 추가
        </button>
      </div>
    </div>
  `;
};

/** -----------------------------
 * 3) Cart 저장소 (localStorage)
 * ------------------------------*/
App.getCart = function () {
  try {
    return JSON.parse(localStorage.getItem("cart")) || [];
  } catch {
    return [];
  }
};

App.setCart = function (cart) {
  localStorage.setItem("cart", JSON.stringify(cart));
};

App.addToCart = function (productId) {
  const products = App.products;
  const cart = App.getCart();

  const product = products.find((p) => p.id === productId);
  if (!product) return;

  const existing = cart.find((item) => item.id === productId);

  if (existing) {
    existing.quantity += 1;
  } else {
    cart.push({
      id: product.id,
      name: product.name,
      price: product.price,
      image: product.image,
      quantity: 1
    });
  }

  App.setCart(cart);
  App.updateCartCount();
};

App.updateCartCount = function () {
  const cart = App.getCart();
  const totalItems = cart.reduce((sum, item) => sum + item.quantity, 0);

  const cartCount = document.querySelector(".cart-count");
  if (cartCount) cartCount.textContent = totalItems;
};
```

### 4.3 `js/products.js` (필터/정렬/검색쿼리 적용 보완)

```javascript
/**
 * products.js
 * - 상품 목록 페이지: 렌더링 + 필터 + 정렬 + 검색 결과 처리

let currentProducts = [...App.products];

document.addEventListener("DOMContentLoaded", () => {

function renderProducts(list) {

  if (!container) return;

function setupProductGridEvents() {
  const container = document.getElementById("productsGrid");
  if (!container) return;

  container.addEventListener("click", (e) => {
    const addBtn = e.target.closest('[data-action="add-to-cart"]');
    if (addBtn) {
      const productId = parseInt(addBtn.dataset.id, 10);
      App.addToCart(productId);
      alert("장바구니에 추가되었습니다!");
      return;
    }

    const card = e.target.closest(".product-card");
    if (card) {
      const productId = card.dataset.id;

/**
 * URL 파라미터 예:
 * - products.html?q=마우스
 * - products.html?category=electronics
 */
function applyQueryFilters() {
  const params = new URLSearchParams(window.location.search);
  const q = params.get("q");
  const category = params.get("category");

  let filtered = [...App.products];

  if (category) {
    filtered = filtered.filter((p) => p.category === category);
    const title = document.getElementById("productsTitle");
    if (title) title.textContent = `카테고리: ${category}`;
  }

  if (q) {
    const keyword = q.toLowerCase();
    filtered = filtered.filter((p) => p.name.toLowerCase().includes(keyword));
    const title = document.getElementById("productsTitle");
    if (title) title.textContent = `검색 결과: "${q}"`;
  }

  currentProducts = filtered;
}

function setupFilters() {
  const checkboxes = document.querySelectorAll(
    '.filter-group input[type="checkbox"]'
  );

  checkboxes.forEach((checkbox) => {
    checkbox.addEventListener("change", applyFilters);
  });

  const priceRange = document.getElementById("priceRange");
  if (priceRange) {
    priceRange.addEventListener("input", (e) => {
      const value = parseInt(e.target.value, 10);
      document.getElementById("priceValue").textContent =
        value.toLocaleString() + "원";
      applyFilters();
    });
  }
}

function applyFilters() {
  const selectedCategories = Array.from(
    document.querySelectorAll('.filter-group input[type="checkbox"]:checked')
  ).map((cb) => cb.value);

  const maxPrice = parseInt(document.getElementById("priceRange").value, 10);

  // ✅ 필터는 일반적으로 "전체 상품" 기준으로 다시 계산
  let filtered = [...App.products];

  // 카테고리 체크박스
  if (selectedCategories.length > 0) {
    filtered = filtered.filter((p) => selectedCategories.includes(p.category));
  }

  // 가격
  filtered = filtered.filter((p) => p.price <= maxPrice);

  currentProducts = filtered;
  renderProducts(currentProducts);
}

function setupSort() {
  const sortSelect = document.getElementById("sortSelect");
  if (!sortSelect) return;

  sortSelect.addEventListener("change", (e) => {
    const value = e.target.value;

    // ✅ 현재 목록(currentProducts)을 기준으로 정렬
    let sorted = [...currentProducts];

    switch (value) {
      case "price-low":
        sorted.sort((a, b) => a.price - b.price);
        break;
      case "price-high":
        sorted.sort((a, b) => b.price - a.price);
        break;
      case "popular":
        sorted.sort((a, b) => b.rating - a.rating);
        break;
      case "newest":
        sorted.sort((a, b) => b.id - a.id);
        break;
    }

    renderProducts(sorted);
  });
}
```

## 5. 장바구니 기능

### 5.1 `cart.js` (보완)

현재 `cart.js`는 `cart` 변수를 전역으로 잡고 있는데, 위에서 `utils.js`를 `App.getCart()`로 바꿨으니 `cart.js`도 그 흐름에 맞추는 게 좋습니다.

✅ 보완: “항상 최신 cart를 읽고 렌더”하면 페이지 새로고침/이동에도 안정적입니다.

```javascript
/**
 * cart.js
 * - 장바구니 페이지 렌더 + 수량 변경 + 삭제 + 합계 계산
 */

document.addEventListener("DOMContentLoaded", () => {
  App.updateCartCount();
  renderCart();
});

function renderCart() {
  const container = document.getElementById("cartItems");
  if (!container) return;

  const cart = App.getCart();

  if (cart.length === 0) {
    container.innerHTML = "<p>장바구니가 비어있습니다.</p>";
    updateTotal(0);
    return;
  }

  container.innerHTML = cart
    .map(
      (item) => `
      <div class="cart-item" data-id="${item.id}">
        <img src="${item.image}" alt="${item.name}" onerror="this.src='images/placeholder.jpg';">
        <div class="cart-item-info">
          <h3>${item.name}</h3>
          <p>${App.formatPrice(item.price)}</p>
        </div>

        <div class="cart-item-controls">
          <button type="button" class="quantity-btn" data-action="decrease" data-id="${item.id}">-</button>
          <span class="quantity">${item.quantity}</span>
          <button type="button" class="quantity-btn" data-action="increase" data-id="${item.id}">+</button>
        </div>

        <div class="cart-item-total">
          ${App.formatPrice(item.price * item.quantity)}
        </div>

        <button type="button" class="remove-btn" data-action="remove" data-id="${item.id}">삭제</button>
      </div>
    `
    )
    .join("");

  // ✅ 이벤트 위임으로 간단/안정적으로 처리
  container.addEventListener("click", handleCartClick, { once: true });

  updateTotalFromCart();
}

function handleCartClick(e) {
  const btn = e.target.closest("button");
  if (!btn) return;

  const action = btn.dataset.action;
  const itemId = parseInt(btn.dataset.id, 10);

  const cart = App.getCart();
  const item = cart.find((i) => i.id === itemId);
  if (!item) return;

  if (action === "increase") item.quantity += 1;

  if (action === "decrease") {
    if (item.quantity > 1) item.quantity -= 1;
  }

  if (action === "remove") {
    const next = cart.filter((i) => i.id !== itemId);
    App.setCart(next);
    App.updateCartCount();
    renderCart();
    return;
  }

  App.setCart(cart);
  App.updateCartCount();
  renderCart();
}

function updateTotalFromCart() {
  const cart = App.getCart();
  const total = cart.reduce((sum, item) => sum + item.price * item.quantity, 0);
  updateTotal(total);
}

function updateTotal(total) {
  const totalElement = document.getElementById("cartTotal");
  if (totalElement) totalElement.textContent = App.formatPrice(total);
}
```

주의: `container.addEventListener(..., { once: true })` 를 사용했기 때문에 `renderCart()`가 다시 실행될 때 이벤트가 중복 등록되지 않습니다(초보자 실수 방지).
