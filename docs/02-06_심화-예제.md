# 심화 예제 - 쇼핑몰 프로젝트
## 📚 학습 목표

이 심화 예제를 완료하면 다음을 할 수 있습니다:

- (실습용) 사용자 인증(회원가입/로그인/로그아웃) 흐름 이해
- 주문 생성 → 주문내역 조회까지 데이터 흐름 구현
- 상품 리뷰 작성/조회 구현
- 관리자 페이지(관리자만 접근) + 상품/주문 관리 기능 구현
- 실무에서 자주 쓰는 패턴(모듈화, 가드, 스토리지 레이어, 이벤트 위임) 적용

⚠️ 중요(실습 범위 안내)  
이 심화 예제는 **백엔드 없이** `localStorage`/`sessionStorage`로만 구현합니다.  
따라서 “진짜 서비스 수준 보안/결제/인증”은 구현할 수 없고, 대신 **흐름을 학습하는 데 집중**합니다.

## 1. 공통 저장소 레이어 만들기(핵심)

실습이 커질수록 “각 파일에서 localStorage를 제각각 다루면” 유지보수가 지옥이 됩니다.  
그래서 `storage.js`로 통일합니다.

### 1.1 `storage.js` (필수)

`js/storage.js`

```javascript
/**
 * storage.js
 * - localStorage/sessionStorage 접근을 한 곳으로 모아두는 파일
 * - 실무에서는 API/DB로 바뀌어도 이 레이어만 교체하면 됩니다.
 */

// 키 이름을 한 곳에서 관리하면 오타로 버그나는 걸 줄일 수 있습니다.
const STORAGE_KEYS = {
  USERS: "users",
  ORDERS: "orders",
  REVIEWS: "reviews",
  CART: "cart",
  CURRENT_USER: "currentUser" // sessionStorage 사용
};

// ---------- 공통 유틸 ----------
function readJSON(storage, key, defaultValue) {
  try {
    const raw = storage.getItem(key);
    return raw ? JSON.parse(raw) : defaultValue;
  } catch (e) {
    console.error("readJSON error:", key, e);
    return defaultValue;
  }
}

function writeJSON(storage, key, value) {
  storage.setItem(key, JSON.stringify(value));
}

// ---------- Users ----------
export function getUsers() {
  return readJSON(localStorage, STORAGE_KEYS.USERS, []);
}

export function saveUsers(users) {
  writeJSON(localStorage, STORAGE_KEYS.USERS, users);
}

// ---------- Auth(Session) ----------
export function getCurrentUser() {
  return readJSON(sessionStorage, STORAGE_KEYS.CURRENT_USER, null);
}

export function setCurrentUser(user) {
  writeJSON(sessionStorage, STORAGE_KEYS.CURRENT_USER, user);
}

export function clearCurrentUser() {
  sessionStorage.removeItem(STORAGE_KEYS.CURRENT_USER);
}

// ---------- Cart ----------
export function getCart() {
  return readJSON(localStorage, STORAGE_KEYS.CART, []);
}

export function saveCart(cart) {
  writeJSON(localStorage, STORAGE_KEYS.CART, cart);
}

// ---------- Orders ----------
export function getOrders() {
  return readJSON(localStorage, STORAGE_KEYS.ORDERS, []);
}

export function saveOrders(orders) {
  writeJSON(localStorage, STORAGE_KEYS.ORDERS, orders);
}

// ---------- Reviews ----------
export function getReviews() {
  return readJSON(localStorage, STORAGE_KEYS.REVIEWS, []);
}

export function saveReviews(reviews) {
  writeJSON(localStorage, STORAGE_KEYS.REVIEWS, reviews);
}

// ---------- 권한 가드 ----------
export function requireLogin(redirectTo = "login.html") {
  const user = getCurrentUser();
  if (!user) {
    alert("로그인이 필요합니다.");
    window.location.href = redirectTo;
    return null;
  }
  return user;
}

export function requireAdmin(redirectTo = "index.html") {
  const user = getCurrentUser();
  if (!user || user.role !== "admin") {
    alert("관리자만 접근할 수 있습니다.");
    window.location.href = redirectTo;
    return null;
  }
  return user;
}
```

참고: 정적 HTML에서 `export`를 쓰려면 `<script type="module">`이 필요합니다.  
아래 문서부터는 **module 방식**으로 설명합니다(현대 프론트엔드 흐름에 더 좋습니다).

## 2. 사용자 인증 기능(회원가입/로그인/로그아웃)

### 2.1 데이터 구조(실습용)

사용자(user)는 최소한 아래 필드를 갖게 합니다.

```json
{
  "id": 1700000000000,
  "email": "test@example.com",
  "password": "1234",
  "name": "홍길동",
  "role": "user"
}
```

⚠️ 보안 안내(문서에 반드시 포함 권장)

- 비밀번호를 localStorage에 저장하는 방식은 실서비스에서 절대 사용하면 안 됩니다.
- 실무에서는 서버에서 해시 처리 + 토큰 기반 인증을 합니다.

### 2.3 로그인/회원가입 로직(`auth.js`)

`js/auth.js`

```javascript
import { getUsers, saveUsers, setCurrentUser } from "./storage.js";

/**
 * auth.js
 * - login.html, signup.html에서 공통으로 사용
 * - 페이지에 어떤 폼이 존재하는지 확인한 뒤 해당 로직만 실행
 */

const loginForm = document.getElementById("loginForm");
const signupForm = document.getElementById("signupForm");

// ---------- 회원가입 ----------
if (signupForm) {
  signupForm.addEventListener("submit", (e) => {
    e.preventDefault();

    const name = document.getElementById("name").value.trim();
    const email = document.getElementById("email").value.trim().toLowerCase();
    const password = document.getElementById("password").value;

    const users = getUsers();

    // 이메일 중복 체크
    const exists = users.some((u) => u.email === email);
    if (exists) {
      alert("이미 가입된 이메일입니다.");
      return;
    }

    const newUser = {
      id: Date.now(),
      name,
      email,
      password, // ⚠️ 실습용
      role: "user"
    };

    users.push(newUser);
    saveUsers(users);

    alert("회원가입 완료! 로그인 페이지로 이동합니다.");
    window.location.href = "login.html";
  });
}

// ---------- 로그인 ----------
if (loginForm) {
  loginForm.addEventListener("submit", (e) => {
    e.preventDefault();

    const email = document.getElementById("email").value.trim().toLowerCase();
    const password = document.getElementById("password").value;

    const users = getUsers();
    const user = users.find((u) => u.email === email && u.password === password);

    if (!user) {
      alert("이메일 또는 비밀번호가 올바르지 않습니다.");
      return;
    }

    // 로그인 성공: 세션에 저장(브라우저 닫으면 로그아웃 효과)
    setCurrentUser({ id: user.id, email: user.email, name: user.name, role: user.role });

    alert("로그인 성공!");
    window.location.href = "index.html";
  });
}
```

✅ 개선 포인트

- 로그인 성공 시 sessionStorage에 저장하는 값은 비밀번호 제외(실습이라도 최소 방어)
- 이메일은 소문자로 통일해서 비교

## 3. 주문 내역 관리(Checkout → Orders)

기존 문서의 가장 큰 문제는 **주문 저장 시 `userId`를 안 넣어서**  
`orders.filter(order => order.userId === currentUser.id)`가 절대 매칭되지 않는다는 점입니다.

### 3.1 `checkout.html`(개념)

- 로그인된 사용자만 접근 가능(`requireLogin`)
- 주문 데이터에 반드시 `userId` 포함
- 주문 완료 후 `order-complete.html`은 선택(없으면 `orders.html`로 이동)

---

### 3.2 `checkout.js` (수정본)

`js/checkout.js`

```javascript
import { requireLogin, getCart, saveCart, getOrders, saveOrders } from "./storage.js";

document.addEventListener("DOMContentLoaded", () => {
  const user = requireLogin(); // 로그인 안 했으면 여기서 이동됨
  if (!user) return;

  const cart = getCart();
  if (cart.length === 0) {
    alert("장바구니가 비어있습니다.");
    window.location.href = "cart.html";
    return;
  }

  const checkoutForm = document.getElementById("checkoutForm");
  checkoutForm.addEventListener("submit", (e) => {
    e.preventDefault();

    // 폼 데이터 수집(예시)
    const orderData = {
      userId: user.id,
      shipping: {
        name: document.getElementById("name").value.trim(),
        phone: document.getElementById("phone").value.trim(),
        address: document.getElementById("address").value.trim(),
        memo: document.getElementById("memo").value.trim()
      },
      paymentMethod: document.querySelector('input[name="payment"]:checked')?.value || "card",
      items: cart,
      total: cart.reduce((sum, item) => sum + item.price * item.quantity, 0)
    };

    processOrder(orderData);
  });
});

function processOrder(orderData) {
  const orders = getOrders();

  const order = {
    id: Date.now(),
    ...orderData,
    date: new Date().toISOString(),
    status: "주문완료"
  };

  orders.push(order);
  saveOrders(orders);

  // 장바구니 비우기
  saveCart([]);

  alert("주문이 완료되었습니다!");
  window.location.href = "orders.html";
}
```

✅ 개선 포인트

- 주문 데이터에 `userId` 포함 (필수)
- 장바구니 비우기도 `saveCart([])`로 통일

### 3.3 `orders.js` (수정본)

`js/orders.js`

```javascript
import { requireLogin, getOrders } from "./storage.js";

document.addEventListener("DOMContentLoaded", () => {
  const user = requireLogin();
  if (!user) return;

  displayOrders(user.id);
});

function displayOrders(userId) {
  const orders = getOrders();
  const userOrders = orders.filter((o) => o.userId === userId).sort((a, b) => b.id - a.id);

  const container = document.getElementById("ordersList");

  if (!userOrders.length) {
    container.innerHTML = "<p>주문 내역이 없습니다.</p>";
    return;
  }

  container.innerHTML = userOrders
    .map(
      (order) => `
      <div class="order-item">
        <div class="order-header">
          <span>주문번호: ${order.id}</span>
          <span>${order.status}</span>
          <span>${new Date(order.date).toLocaleDateString()}</span>
        </div>

        <div class="order-items">
          ${order.items
            .map(
              (item) => `
            <div class="order-item-detail">
              <img src="${item.image}" alt="${item.name}" />
              <div>
                <h4>${item.name}</h4>
                <p>${item.quantity}개 × ${item.price.toLocaleString()}원</p>
              </div>
            </div>
          `
            )
            .join("")}
        </div>

        <div class="order-total">
          총 금액: ${order.total.toLocaleString()}원
        </div>
      </div>
    `
    )
    .join("");
}
```

### 4.2 `reviews.js` (작성/표시 로직)

`js/reviews.js`

```javascript
import { requireLogin, getReviews, saveReviews } from "./storage.js";

/**
 * 사용 위치:
 * - product-detail.html에서 productId를 얻은 뒤
 *   renderReviews(productId), setupReviewForm(productId) 호출
 */

export function renderReviews(productId) {
  const container = document.getElementById("reviewsList");
  const reviews = getReviews().filter((r) => r.productId === productId).sort((a, b) => b.id - a.id);

  if (!reviews.length) {
    container.innerHTML = "<p>아직 리뷰가 없습니다. 첫 리뷰를 남겨보세요!</p>";
    return;
  }

  container.innerHTML = reviews
    .map(
      (r) => `
      <div class="review-item">
        <div class="review-meta">
          <strong>${r.userName}</strong>
          <span>평점: ${"⭐".repeat(r.rating)}</span>
          <span>${new Date(r.date).toLocaleDateString()}</span>
        </div>
        <p>${escapeHTML(r.comment)}</p>
      </div>
    `
    )
    .join("");
}

export function setupReviewForm(productId) {
  const form = document.getElementById("reviewForm");
  if (!form) return;

  form.addEventListener("submit", (e) => {
    e.preventDefault();

    const user = requireLogin();
    if (!user) return;

    const rating = parseInt(document.getElementById("rating").value, 10);
    const comment = document.getElementById("comment").value.trim();

    if (!comment) {
      alert("리뷰 내용을 입력해주세요.");
      return;
    }

    const reviews = getReviews();
    const newReview = {
      id: Date.now(),
      productId,
      userId: user.id,
      userName: user.name,
      rating,
      comment,
      date: new Date().toISOString()
    };

    reviews.push(newReview);
    saveReviews(reviews);

    alert("리뷰가 등록되었습니다!");
    form.reset();
    renderReviews(productId);
  });
}

// XSS 방지(실습에서도 최소한의 안전장치)
function escapeHTML(str) {
  return str
    .replaceAll("&", "&amp;")
    .replaceAll("<", "&lt;")
    .replaceAll(">", "&gt;")
    .replaceAll('"', "&quot;")
    .replaceAll("'", "&#039;");
}
```

✅ 개선 포인트

- 리뷰 출력 시 HTML 이스케이프(실무에서 중요한 습관)
- 로그인 강제(`requireLogin`)

## 5. 관리자 페이지(권한 가드 포함)

관리자 기능은 “UI만” 있으면 실습이 아니라 “목록”이 됩니다.  
그래서 최소한 아래 2가지는 반드시 들어가야 합니다.

- 관리자만 접근 가능한 가드(`requireAdmin`)
- 상품/주문 데이터를 실제로 수정할 수 있는 CRUD(실습용)

### 5.1 관리자 계정 만들기(실습 편의)

가장 쉬운 방법: signup 후 localStorage에서 `role`을 `admin`으로 바꾸거나, 초기 시드로 주입.

초기 시드(선택): `auth.js` 또는 별도 `seed.js`에서 1회 실행

```javascript
import { getUsers, saveUsers } from "./storage.js";

const users = getUsers();
const adminExists = users.some((u) => u.email === "admin@shop.com");

if (!adminExists) {
  users.push({
    id: Date.now(),
    name: "관리자",
    email: "admin@shop.com",
    password: "admin1234",
    role: "admin"
  });
  saveUsers(users);
}
```

문서에는 `admin@shop.com / admin1234`로 로그인해보라고 안내하면 실습이 매끄럽습니다.

---

### 5.2 `admin.js` (가드 + 대시보드 최소 구현)

`js/admin.js`

```javascript
import { requireAdmin, getOrders } from "./storage.js";

document.addEventListener("DOMContentLoaded", () => {
  const admin = requireAdmin();
  if (!admin) return;

  renderDashboard();
});

function renderDashboard() {
  const orders = getOrders();

  const totalOrders = orders.length;
  const totalRevenue = orders.reduce((sum, o) => sum + (o.total || 0), 0);

  document.getElementById("totalOrders").textContent = totalOrders;
  document.getElementById("totalRevenue").textContent = totalRevenue.toLocaleString() + "원";
}
```

## 6. 고급 기능(문서 품질을 올리는 방식으로 정리)

기존 문서의 “실시간 검색/무한스크롤/지연로딩”은 좋지만,  
초보자 실습에서는 예외처리와 성능 안전장치를 함께 넣어주는 게 중요합니다.

### 6.1 실시간 검색(디바운스 추가)

키 입력마다 필터를 돌리면 데이터가 많아질수록 버벅입니다.  
디바운스를 추가하면 실무 패턴을 자연스럽게 학습합니다.

```javascript
function debounce(fn, delay = 200) {
  let timer;
  return (...args) => {
    clearTimeout(timer);
    timer = setTimeout(() => fn(...args), delay);
  };
}

const searchInput = document.getElementById("searchInput");
if (searchInput) {
  searchInput.addEventListener(
    "input",
    debounce((e) => {
      const query = e.target.value.toLowerCase();
      const results = products.filter((p) =>
        (p.name || "").toLowerCase().includes(query)
      );
      displaySearchResults(results);
    }, 250)
  );
}
```

### 6.2 무한 스크롤(중복 로드 방지 + 바닥 여유)

```javascript
let currentPage = 1;
const itemsPerPage = 12;
let isLoading = false;

window.addEventListener("scroll", () => {
  const nearBottom = window.innerHeight + window.scrollY >= document.body.offsetHeight - 200;
  if (nearBottom) loadMoreProducts();
});

function loadMoreProducts() {
  if (isLoading) return;
  isLoading = true;

  currentPage++;
  const start = (currentPage - 1) * itemsPerPage;
  const end = start + itemsPerPage;
  const more = products.slice(start, end);

  if (more.length === 0) return; // 더 이상 없음

  appendProducts(more);

  // 한 프레임 뒤에 로딩 해제(연속 호출 방지)
  requestAnimationFrame(() => {
    isLoading = false;
  });
}
```

### 6.3 이미지 지연 로딩(실전 팁 포함)

`loading="lazy"`가 지원되는 브라우저가 많습니다.  
IntersectionObserver는 “지원/커스터마이징” 목적일 때 사용합니다.

```html
<img
  src="images/placeholder.jpg"
  data-src="images/products/mouse.jpg"
  loading="lazy"
  alt="상품"
/>
```

```javascript
const images = document.querySelectorAll('img[data-src]');
const imageObserver = new IntersectionObserver((entries, obs) => {
  entries.forEach((entry) => {
    if (!entry.isIntersecting) return;

    const img = entry.target;
    img.src = img.dataset.src;

    // 로딩 실패 대비(플레이스홀더)
    img.onerror = () => {
      img.src = "images/placeholder.jpg";
    };

    obs.unobserve(img);
  });
});

images.forEach((img) => imageObserver.observe(img));
```

## 7. 성능/보안 파트(문장 보강)

### 7.1 localStorage 용량 제한

- localStorage는 브라우저마다 다르지만 대략 수 MB 수준이라 데이터가 커지면 위험
- 이미지/대용량 배열/로그를 무작정 쌓지 말기
- 주문/리뷰가 커지면 “페이지네이션” 개념을 도입(다음 단계)

### 7.2 보안(실습용 경고 문구를 더 강하게)

- 비밀번호/결제/개인정보는 클라이언트 저장 금지
- 실무에서는:
  - 서버 인증(해시)
  - HTTPS
  - 토큰/세션
  - 권한 체크(서버 측)
